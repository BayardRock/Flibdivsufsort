open Flibdivsufsort
open System.Text

let makeStringSuffixArray (text: string) =    
    new SuffixArray(Encoding.ASCII.GetBytes(text.ToUpperInvariant()))

let searchStringSuffixArray (pattern: string) (sa: SuffixArray) =
    sa.Search(Encoding.ASCII.GetBytes(pattern))
